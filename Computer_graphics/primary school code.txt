// ---------------- SUN ----------------
void drawSun() {
    float centerX = 350;
    float centerY = 400;
    float radius = 35;
    int rays = 12;

    glColor3f(1.0f, 0.87f, 0.13f);
    glBegin(GL_TRIANGLE_FAN);
    glVertex2f(centerX, centerY);
    for (int i = 0; i <= 100; i++) {
        float angle = 2.0f * 3.1415926f * i / 100;
        glVertex2f(centerX + cos(angle) * radius,
                   centerY + sin(angle) * radius);
    }
    glEnd();

    glLineWidth(3);
    glColor3f(1.0f, 0.9f, 0.2f);
    glBegin(GL_LINES);
    float rayExtension = 25.0f * lightIntensity;
    for (int i = 0; i < rays; i++) {
        float angle = 2.0f * 3.1415926f * i / rays;
        float x1 = centerX + cos(angle) * (radius + 5);
        float y1 = centerY + sin(angle) * (radius + 5);
        float x2 = centerX + cos(angle) * (radius + 5 + rayExtension);
        float y2 = centerY + sin(angle) * (radius + 5 + rayExtension);
        glVertex2f(x1, y1);
        glVertex2f(x2, y2);
    }
    glEnd();
}

// ---------------- SCHOOL (3 FLOORS WITH KIDS IN WINDOWS) ----------------
void drawSchool() {
    int baseLeft = -180, baseRight = 280;
    int baseBottom = 0, baseTop = 280; // 3 floors

    // Main body
    glColor3f(0.05f, 0.1f, 0.45f);
    glBegin(GL_POLYGON);
    glVertex2f(baseLeft,  baseBottom);
    glVertex2f(baseRight, baseBottom);
    glVertex2f(baseRight, baseTop);
    glVertex2f(baseLeft,  baseTop);
    glEnd();

    // Header band
    glColor3f(0.95f, 0.95f, 0.2f);
    glBegin(GL_POLYGON);
    glVertex2f(baseLeft - 20, baseTop);
    glVertex2f(baseRight + 20, baseTop);
    glVertex2f(baseRight + 20, baseTop + 50);
    glVertex2f(baseLeft - 20, baseTop + 50);
    glEnd();

    // Door (ground floor center)
    glColor3f(1, 1, 1);
    glBegin(GL_POLYGON);
    glVertex2f(30, baseBottom);
    glVertex2f(70, baseBottom);
    glVertex2f(70, baseBottom + 90);
    glVertex2f(30, baseBottom + 90);
    glEnd();

    // Windows grid: 3 floors x 4 columns
    int floors = 3, cols = 4;
    int floorHeight = 80;
    int windowW = 40, windowH = 30;
    int xStart = baseLeft + 20;
    int xGap = ((baseRight - baseLeft) - 40 - cols*windowW) / (cols - 1); // spacing

    glColor3f(0.9f, 0.95f, 1.0f);
    for (int f = 0; f < floors; f++) {
        int y = baseBottom + 30 + f * (floorHeight + 10);
        for (int c = 0; c < cols; c++) {
            int x = xStart + c * (windowW + xGap);
            // Window
            glBegin(GL_POLYGON);
            glVertex2f(x,         y);
            glVertex2f(x+windowW, y);
            glVertex2f(x+windowW, y+windowH);
            glVertex2f(x,         y+windowH);
            glEnd();
            // Frame detail
            glColor3f(0.75f, 0.85f, 1.0f);
            glBegin(GL_LINE_LOOP);
            glVertex2f(x,         y);
            glVertex2f(x+windowW, y);
            glVertex2f(x+windowW, y+windowH);
            glVertex2f(x,         y+windowH);
            glEnd();

            // Kids in some windows
            glColor3f(0.9f, 0.95f, 1.0f);
            bool placeKid = ((f + c) % 2 == 0) || (f == 2 && (c == 1 || c == 2));
            if (placeKid) {
                drawKidWindow(x + windowW/2.0f, y + windowH/2.0f - 4, (c % 2) == 0);
            }
            glColor3f(0.9f, 0.95f, 1.0f);
        }
    }

    // Sign text
    glColor3f(0.0f, 0.0f, 0.0f);
    const char* text = "USD PRIMARY SCHOOL";
    float width = 0;
    for (int i = 0; i < (int)strlen(text); i++)
        width += glutBitmapWidth(GLUT_BITMAP_HELVETICA_18, text[i]);
    float textX = baseLeft - 20 + ((baseRight + 20 - (baseLeft - 20)) - width) / 2.0f;
    glRasterPos2f(textX, baseTop + 20);
    for (int i = 0; i < (int)strlen(text); i++)
        glutBitmapCharacter(GLUT_BITMAP_HELVETICA_18, text[i]);
}

// ---------------- BIRD (WING FLAPPING) ----------------
void drawBird(float x, float y, float wingPhase) {
    // Body (small oval)
    glColor3f(0.1f, 0.1f, 0.1f);
    glBegin(GL_POLYGON);
    for (int i = 0; i < 360; i++) {
        float angle = i * 3.1415926f / 180.0f;
        glVertex2f(x + cos(angle) * 11, y + sin(angle) * 6);
    }
    glEnd();

    // Wings (flapping with sine)
    float flap = sin(wingPhase) * 12.0f + 6.0f; // amplitude + base
    glLineWidth(3);
    glColor3f(0.15f, 0.15f, 0.15f);
    glBegin(GL_LINES);
    // Left wing
    glVertex2f(x - 5, y + 2);
    glVertex2f(x - flap, y + flap * 0.6f);
    // Right wing
    glVertex2f(x + 5, y + 2);
    glVertex2f(x + flap, y + flap * 0.6f);
    glEnd();

    // Beak
    glColor3f(0.9f, 0.7f, 0.1f);
    glBegin(GL_TRIANGLES);
    glVertex2f(x + 12, y);
    glVertex2f(x + 18, y + 2);
    glVertex2f(x + 12, y + 4);
    glEnd();
}

// ---------------- CLOUD ----------------
void drawCloud(float x, float y) {
    glColor3f(0.78f, 0.78f, 0.78f);

    glBegin(GL_POLYGON);
    for (int i = 0; i < 360; i++)
        glVertex2f(x + cos(i * 3.14159f / 180.0f) * 50,
                   y + sin(i * 3.14159f / 180.0f) * 25);
    glEnd();

    glBegin(GL_POLYGON);
    for (int i = 0; i < 360; i++)
        glVertex2f(x + 60 + cos(i * 3.14159f / 180.0f) * 50,
                   y + 10 + sin(i * 3.14159f / 180.0f) * 25);
    glEnd();

    glBegin(GL_POLYGON);
    for (int i = 0; i < 360; i++)
        glVertex2f(x - 60 + cos(i * 3.14159f / 180.0f) * 50,
                   y + 20 + sin(i * 3.14159f / 180.0f) * 25);
    glEnd();
}

// ---------------- STUDENT (HUMAN-LIKE) ----------------
void drawStudent(float x, float y, bool facingRight, float armSwingPhase,
                 float shirtR=0.1f, float shirtG=0.4f, float shirtB=0.9f) {
    // Head
    glColor3f(0.95f, 0.8f, 0.6f);
    glBegin(GL_POLYGON);
    for (int i = 0; i < 360; i++) {
        float a = i * 3.1415926f / 180.0f;
        glVertex2f(x + cos(a) * 6, y + 20 + sin(a) * 6);
    }
    glEnd();

    // Shirt
    glColor3f(shirtR, shirtG, shirtB);
    glBegin(GL_POLYGON);
    glVertex2f(x - 7, y + 6);
    glVertex2f(x + 7, y + 6);
    glVertex2f(x + 7, y - 8);
    glVertex2f(x - 7, y - 8);
    glEnd();

    // Pants
    glColor3f(0.1f, 0.1f, 0.1f);
    glBegin(GL_POLYGON);
    glVertex2f(x - 7, y - 8);
    glVertex2f(x + 7, y - 8);
    glVertex2f(x + 7, y - 18);
    glVertex2f(x - 7, y - 18);
    glEnd();

    // Arms (swing)
    float swing = sin(armSwingPhase) * 6.0f;
    glLineWidth(2);
    glColor3f(0.95f, 0.8f, 0.6f);
    glBegin(GL_LINES);
    glVertex2f(x - 7, y + 4);
    glVertex2f(x - 15, y + 4 + swing);
    glVertex2f(x + 7, y + 4);
    glVertex2f(x + 15, y + 4 - swing);
    glEnd();

    // Legs (walking)
    glBegin(GL_LINES);
    glVertex2f(x - 3, y - 18);
    glVertex2f(x - 3, y - 28 - swing * 0.3f);
    glVertex2f(x + 3, y - 18);
    glVertex2f(x + 3, y - 28 + swing * 0.3f);
    glEnd();

    // Eyes
    glPointSize(3);
    glColor3f(0, 0, 0);
    glBegin(GL_POINTS);
    float eyeOffset = facingRight ? 2.0f : -2.0f;
    glVertex2f(x + eyeOffset, y + 21);
    glVertex2f(x + eyeOffset + (facingRight ? 3.0f : -3.0f), y + 21);
    glEnd();
}

// ---------------- BACKPACK KID (ENTERING SCHOOL) ----------------
void drawBackpackKid(float t) {
    // Move along walkway center line from road to door
    // Parametric interpolation between bottom midpoint and top midpoint of walkway
    float xStart = (walkway[2].x + walkway[3].x) * 0.5f;
    float yStart = (walkway[2].y + walkway[3].y) * 0.5f;
    float xEnd   = (walkway[0].x + walkway[1].x) * 0.5f;
    float yEnd   = (walkway[0].y + walkway[1].y) * 0.5f;

    float x = xStart + (xEnd - xStart) * t;
    float y = yStart + (yEnd - yStart) * t;

    // Draw kid with backpack slightly turned toward school (facingRight ~ true near door)
    bool facingRight = true;
    float swingPhase = t * 6.28318f * 2.0f;

    // Kid body
    drawStudent(x, y, facingRight, swingPhase, 0.15f, 0.45f, 0.85f);

    // Backpack (rectangle behind body on left shoulder)
    glColor3f(0.25f, 0.20f, 0.10f);
    glBegin(GL_POLYGON);
    glVertex2f(x - 9, y + 4);
    glVertex2f(x + 1, y + 4);
    glVertex2f(x + 1, y - 10);
    glVertex2f(x - 9, y - 10);
    glEnd();

    // Strap
    glColor3f(0.18f, 0.15f, 0.08f);
    glLineWidth(2);
    glBegin(GL_LINES);
    glVertex2f(x + 1, y + 4);
    glVertex2f(x + 7, y - 6);
    glEnd();
}

// ---------------- DISPLAY ----------------
void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    // Background first
    drawSun();
    drawGround();
    drawMountains();

    // Clouds BEHIND buildings
    drawCloud(cloud1X, 360);
    drawCloud(cloud2X, 420);

    // Foreground environment
    drawHouse();
    drawSchool();

    drawTree(-480, 0);
    drawTree(-200, 0);
    drawTree(80,  0);
    drawTree(380, 0);

    drawFieldsAndFence(); // Fields + smaller fence (natural)
    drawWalkway();        // Walkway from door to road

    drawRoad();

    // Street lights
    drawStreetLight(-400);
    drawStreetLight(-200);
    drawStreetLight(0);
    drawStreetLight(200);
    drawStreetLight(400);

    // Cars
    drawCar(car1X, carY, 1, 0, 0);
    drawCar(car2X, carY, 0, 1, 0);

    // Birds with flapping wings
    for (int i = 0; i < birdCount; i++) {
        drawBird(birds[i].x, birds[i].y, birds[i].wingPhase);
    }

    // Kids playing safely INSIDE fields (left side of walkway)
    for (int i = 0; i < leftKidsCount; i++) {
        drawStudent(leftFieldKids[i].x, leftFieldKids[i].y,
                    leftFieldKids[i].facingRight, leftFieldKids[i].armSwing,
                    leftFieldKids[i].shirtR, leftFieldKids[i].shirtG, leftFieldKids[i].shirtB);
    }
    // Kids playing safely INSIDE fields (right side of walkway)
    for (int i = 0; i < rightKidsCount; i++) {
        drawStudent(rightFieldKids[i].x, rightFieldKids[i].y,
                    rightFieldKids[i].facingRight, rightFieldKids[i].armSwing,
                    rightFieldKids[i].shirtR, rightFieldKids[i].shirtG, rightFieldKids[i].shirtB);
    }

    // Backpack kid entering school along walkway
    drawBackpackKid(bpKid.t);

    glutSwapBuffers();
}

// ---------------- TIMER ----------------
void timer(int) {
    // Sun light flicker
    if (decreasing) lightIntensity -= 0.05f;
    else            lightIntensity += 0.05f;
    if (lightIntensity <= 0.2f) decreasing = false;
    if (lightIntensity >= 1.0f) decreasing = true;

    // Cars
    car1X += 5;
    car2X -= 5;
    if (car1X > 600)  car1X = -600;
    if (car2X < -600) car2X = 600;

    // Clouds move slowly (behind buildings)
    cloud1X += cloudSpeed;
    cloud2X += cloudSpeed;
    if (cloud1X > 600) cloud1X = -600;
    if (cloud2X > 600) cloud2X = -600;

    // Birds wings flapping and forward movement
    for (int i = 0; i < birdCount; i++) {
        birds[i].x += birds[i].vx;
        birds[i].wingPhase += 0.25f + 0.05f * i; // slightly different rate
        if (birds[i].wingPhase > 6.28318f) birds[i].wingPhase -= 6.28318f;
        if (birds[i].x > 520) birds[i].x = -520; // loop around
    }

    // Kids move within left field bounds (stay inside fence area)
    float leftBound = fieldLeftX + 20;
    float rightBound = (walkway[3].x + walkway[2].x) * 0.5f - 15; // near walkway center minus margin
    for (int i = 0; i < leftKidsCount; i++) {
        leftFieldKids[i].x += leftFieldKids[i].vx;
        leftFieldKids[i].armSwing += 0.2f;
        if (leftFieldKids[i].armSwing > 6.28318f) leftFieldKids[i].armSwing -= 6.28318f;

        if (leftFieldKids[i].x < leftBound) {
            leftFieldKids[i].x = leftBound;
            leftFieldKids[i].vx = fabs(leftFieldKids[i].vx);
            leftFieldKids[i].facingRight = true;
        } else if (leftFieldKids[i].x > rightBound) {
            leftFieldKids[i].x = rightBound;
            leftFieldKids[i].vx = -fabs(leftFieldKids[i].vx);
            leftFieldKids[i].facingRight = false;
        }
        // Clamp Y inside field
        if (leftFieldKids[i].y < fieldBottomY + 10) leftFieldKids[i].y = fieldBottomY + 10;
        if (leftFieldKids[i].y > fieldTopY - 10)    leftFieldKids[i].y = fieldTopY - 10;
    }

    // Kids move within right field bounds
    float leftBoundR = (walkway[3].x + walkway[2].x) * 0.5f + 15; // near walkway center plus margin
    float rightBoundR = fieldRightX - 20;
    for (int i = 0; i < rightKidsCount; i++) {
        rightFieldKids[i].x += rightFieldKids[i].vx;
        rightFieldKids[i].armSwing += 0.2f;
        if (rightFieldKids[i].armSwing > 6.28318f) rightFieldKids[i].armSwing -= 6.28318f;

        if (rightFieldKids[i].x < leftBoundR) {
            rightFieldKids[i].x = leftBoundR;
            rightFieldKids[i].vx = fabs(rightFieldKids[i].vx);
            rightFieldKids[i].facingRight = true;
        } else if (rightFieldKids[i].x > rightBoundR) {
            rightFieldKids[i].x = rightBoundR;
            rightFieldKids[i].vx = -fabs(rightFieldKids[i].vx);
            rightFieldKids[i].facingRight = false;
        }
        // Clamp Y inside field
        if (rightFieldKids[i].y < fieldBottomY + 10) rightFieldKids[i].y = fieldBottomY + 10;
        if (rightFieldKids[i].y > fieldTopY - 10)    rightFieldKids[i].y = fieldTopY - 10;
    }

    // Backpack kid progresses along walkway towards the door
    if (bpKid.entering) {
        bpKid.t += 0.01f; // speed along walkway
        if (bpKid.t >= 1.0f) {
            bpKid.t = 0.0f; // reset to start, re-enter for continuous animation
        }
    }

    glutPostRedisplay();
    glutTimerFunc(50, timer, 0);
}

// ---------------- MAIN ----------------
int main(int argc, char** argv) {
    glutInit(&argc, argv);
    // Use double buffering for smooth animation
    glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB);
    glutInitWindowSize(900, 700);
    glutCreateWindow("USD Primary School - Safe Fields, Walkway, Kids, Natural Birds");

    init();
    glutDisplayFunc(display);
    glutTimerFunc(50, timer, 0);

    glutMainLoop();
    return 0;
}
